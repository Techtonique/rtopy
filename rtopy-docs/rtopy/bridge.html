 62    def call(
 63        self, r_code: str, r_func: str, return_type: str = "auto", **kwargs
 64    ) -> Any:
 65        """
 66        Call R function with automatic type conversion.
 67
 68        Parameters
 69        ----------
 70        r_code : str
 71            R code defining the function (can include library() calls)
 72        r_func : str
 73            Function name to call
 74        return_type : str
 75            Output type: "auto", "int", "float", "str", "bool",
 76            "list", "dict", "numpy", "pandas", "raw"
 77        **kwargs
 78            Arguments passed to R function
 79
 80        Returns
 81        -------
 82        Result converted to requested Python type
 83
 84        Raises
 85        ------
 86        RNotFoundError
 87            If R is not installed or not in PATH
 88        RExecutionError
 89            If R script fails to execute
 90        RTypeError
 91            If type conversion fails
 92
 93        Examples
 94        --------
 95        >>> rb = RBridge()
 96        >>>
 97        >>> # Simple calculation
 98        >>> code = "add <- function(x, y) x + y"
 99        >>> rb.call(code, "add", x=5, y=3)
100        8.0
101        >>>
102        >>> # Statistical summary
103        >>> code = '''
104        ... summarize <- function(x) {
105        ...     list(mean=mean(x), sd=sd(x), n=length(x))
106        ... }
107        ... '''
108        >>> rb.call(code, "summarize", return_type="dict", x=[1,2,3,4,5])
109        {'mean': 3.0, 'sd': 1.58..., 'n': 5}
110        """
111        if r_func not in r_code:
112            raise ValueError(f"Function '{r_func}' not found in r_code")
113
114        # Convert Python inputs to R-compatible format
115        r_args = self._serialize_args(kwargs)
116
117        # Build and execute R script
118        r_script = self._build_script(r_code, r_func, r_args)
119        output = self._execute_r(r_script)
120
121        # Parse and convert output
122        try:
123            parsed = json.loads(output)
124        except json.JSONDecodeError as e:
125            raise RExecutionError(f"Invalid JSON from R: {output[:200]}") from e
126
127        return self._convert_output(parsed, return_type)

129    def _serialize_args(self, kwargs: Dict) -> str:
130        """Convert Python args to R-compatible JSON."""
131        converted = {}
132
133        for k, v in kwargs.items():
134            if HAS_NUMPY and isinstance(v, np.ndarray):
135                converted[k] = v.tolist()
136            elif HAS_PANDAS and isinstance(v, pd.DataFrame):
137                converted[k] = v.to_dict("list")
138            elif HAS_PANDAS and isinstance(v, pd.Series):
139                converted[k] = v.tolist()
140            else:
141                converted[k] = v
142
143        json_str = json.dumps(converted)
144        return json_str.replace("\\", "\\\\").replace("'", "\\'")

146    def _build_script(self, r_code: str, r_func: str, r_args: str) -> str:
147        """Build R script with error handling."""
148        return f"""
149suppressPackageStartupMessages({{
150    {r_code.strip()}
151    
152    args <- jsonlite::fromJSON('{r_args}')
153    
154    result <- tryCatch(
155        do.call({r_func}, args),
156        error = function(e) stop("R error in {r_func}: ", e$message)
157    )
158    
159    json_out <- jsonlite::toJSON(
160        result,
161        auto_unbox = TRUE,
162        force = TRUE,
163        digits = 15,
164        null = "null",
165        na = "null",
166        dataframe = "columns"
167    )
168    
169    cat(json_out, "\\n")
170}})
171"""

173    def _execute_r(self, script: str) -> str:
174        """Execute R script and return stdout."""
175        with tempfile.NamedTemporaryFile(
176            mode="w", suffix=".R", delete=False, encoding="utf-8"
177        ) as f:
178            f.write(script)
179            temp_file = f.name
180
181        try:
182            proc = subprocess.run(
183                ["Rscript", "--vanilla", temp_file],
184                capture_output=True,
185                text=True,
186                timeout=self.timeout,
187            )
188
189            if self.verbose and proc.stderr:
190                print(f"[R messages] {proc.stderr}")
191
192            if proc.returncode != 0:
193                error_msg = proc.stderr or proc.stdout
194                raise RExecutionError(f"R script failed:\n{error_msg}")
195
196            if not proc.stdout.strip():
197                raise RExecutionError("R produced no output")
198
199            return proc.stdout.strip()
200
201        except subprocess.TimeoutExpired:
202            raise RExecutionError(
203                f"R execution timed out after {self.timeout}s"
204            )
205        finally:
206            try:
207                os.unlink(temp_file)
208            except Exception:
209                pass

211    def _convert_output(self, parsed: Any, return_type: str) -> Any:
212        """Convert parsed JSON to requested Python type."""
213        if return_type == "raw":
214            return parsed
215
216        if return_type == "auto":
217            return_type = self._infer_type(parsed)
218
219        converters = {
220            "int": self._to_int,
221            "float": self._to_float,
222            "str": self._to_str,
223            "bool": self._to_bool,
224            "list": self._to_list,
225            "dict": self._to_dict,
226            "numpy": self._to_numpy,
227            "pandas": self._to_pandas,
228        }
229
230        if return_type not in converters:
231            raise ValueError(
232                f"Unknown return_type '{return_type}'. "
233                f"Valid options: {', '.join(converters.keys())}, raw, auto"
234            )
235
236        try:
237            return converters[return_type](parsed)
238        except Exception as e:
239            raise RTypeError(
240                f"Failed to convert to {return_type}: {str(e)}"
241            ) from e

243    def _infer_type(self, parsed: Any) -> str:
244        """Automatically infer best return type."""
245        if isinstance(parsed, dict):
246            # Check if it looks like a dataframe (dict of lists)
247            if all(isinstance(v, list) for v in parsed.values()):
248                return "pandas" if HAS_PANDAS else "dict"
249            return "dict"
250        elif isinstance(parsed, list):
251            # Check if it's a numeric array
252            if all(isinstance(x, (int, float)) for x in parsed):
253                return "numpy" if HAS_NUMPY else "list"
254            return "list"
255        elif isinstance(parsed, bool):
256            return "bool"
257        elif isinstance(parsed, int):
258            return "int"
259        elif isinstance(parsed, float):
260            return "float"
261        elif isinstance(parsed, str):
262            return "str"
263        return "raw"

307    def _to_numpy(self, val: Any):
308        """Convert to NumPy array."""
309        if not HAS_NUMPY:
310            raise RTypeError(
311                "NumPy not installed. Install with: pip install numpy"
312            )
313
314        if isinstance(val, list):
315            # Handle matrix (list of lists)
316            if val and isinstance(val[0], list):
317                return np.array(val)
318            return np.array(val)
319        if isinstance(val, dict):
320            # Try to convert dict of lists to 2D array
321            lists = list(val.values())
322            if all(isinstance(v, list) for v in lists):
323                return np.array(lists).T
324            return np.array(list(val.values()))
325        return np.array([val])
326
327    def _to_pandas(self, val: Any):
328        """Convert to pandas DataFrame or Series."""
329        if not HAS_PANDAS:
330            raise RTypeError(
331                "pandas not installed. Install with: pip install pandas"
332            )
333
334        if isinstance(val, dict):
335            # Dict of lists -> DataFrame
336            if all(isinstance(v, list) for v in val.values()):
337                return pd.DataFrame(val)
338            # Dict of scalars -> Series
339            return pd.Series(val)
340        if isinstance(val, list):
341            # Check if matrix (list of lists)
342            if val and isinstance(val[0], list):
343                return pd.DataFrame(val)
344            return pd.Series(val)
345        raise RTypeError(f"Cannot convert {type(val).__name__} to pandas")
